usage: PubmedFetch
 -b,--batchSize <NUMBER>                number of records to fetch per
                                        batch
 -h,--help                              Help Message
 -m,--email <EMAIL_ADDRESS>             your contact email address
 -n,--numRecords <NUMBER>               maximum records to return
 -o,--output <CONFIG_FILE>              RecordHandler config file path
 -O,--outputOverride <RH_PARAM=VALUE>   override the RH_PARAM of output
                                        recordhandler using VALUE
 -t,--termSearch <SEARCH_STRING>        term to search against pubmed
                                        repository
 -w,--wordiness <LOG_LEVEL>             Set the console log level
 -X,--config <CONFIG_FILE>              XML Configuration File
